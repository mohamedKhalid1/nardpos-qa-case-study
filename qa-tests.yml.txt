name: QA Automation

on:
  push:
    branches: [main, develop]

jobs:
  run-tests:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v3
      - name: Setup Node
        uses: actions/setup-node@v3
        with:
          node-version: 18
      - run: npm ci
      - run: npm test
      - name: Upload Report
        uses: actions/upload-artifact@v4
        with:
          name: test-report
          path: ./reports
